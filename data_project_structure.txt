# src/data_preprocessing.py
def clean_orders(df):
    df = df.drop_duplicates()
    df['order_date'] = pd.to_datetime(df['order_date'])
    return df

# notebook
from src.data_preprocessing import clean_orders
df = clean_orders(df)

📌 Rule:
If you copy-paste code → move it to src/
